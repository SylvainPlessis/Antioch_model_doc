\section{Thermo data}
\label{data-thermo}

Relations are:
\[
\frac{\enthZ(T)}{\Rg T} = - \tc{0} T^{-2} 
                       + \tc{1} T^{-1} \ln(T)
                       + \tc{2} 
                       + \tc{3} \frac{T}{2}
                       + \tc{4} \frac{T^2}{3}
                       + \tc{5} \frac{T^3}{4}
                       + \tc{6} \frac{T^4}{5}
                       + \tc{8} T^{-1}
\]
\[
\frac{\entrZ(T)}{\Rg}    = - \tc{0} \frac{T^{-2}}{2}
                       - \tc{1} T^{-1}
                       + \tc{2} \ln(T)
                       + \tc{3} T
                       + \tc{4} \frac{T^2}{2}
                       + \tc{5} \frac{T^3}{3}
                       + \tc{6} \frac{T^4}{4}
                       + \tc{9}
\]
